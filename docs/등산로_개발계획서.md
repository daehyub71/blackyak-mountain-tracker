from datetime import datetime

def generate_gpx(trail_name: str, points: list[dict]) -> str:
    """trail_pointsë¥¼ GPX XMLë¡œ ë³€í™˜"""

{elev_tag}      </trkpt>

  </trk>
</gpx>'''
